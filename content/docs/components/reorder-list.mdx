## Usage

### Basic Usage

#### Import the component

```tsx
import { ReorderList } from '@/registry/new-york/components/reorder-list';
```

#### Default Usage

```tsx
<ReorderList>
    <Item variant={'outline'} size={'sm'}>
        <ItemContent>
            <ItemTitle>Reorder List Item 1</ItemTitle>
        </ItemContent>
    </Item>
    <Item variant={'outline'} size={'sm'}>
        <ItemContent>
            <ItemTitle>Reorder List Item 2</ItemTitle>
        </ItemContent>
    </Item>
    <Item variant={'outline'} size={'sm'}>
        <ItemContent>
            <ItemTitle>Reorder List Item 3</ItemTitle>
        </ItemContent>
    </Item>
</ReorderList>
```

#### With Drag Handle

```tsx
<ReorderList withDragHandle>
    <Item variant={'outline'} size={'sm'}>
        <ItemContent>
            <ItemTitle>Reorder List Item 1</ItemTitle>
        </ItemContent>
    </Item>
    <Item variant={'outline'} size={'sm'}>
        <ItemContent>
            <ItemTitle>Reorder List Item 2</ItemTitle>
        </ItemContent>
    </Item>
    <Item variant={'outline'} size={'sm'}>
        <ItemContent>
            <ItemTitle>Reorder List Item 3</ItemTitle>
        </ItemContent>
    </Item>
</ReorderList>
```
