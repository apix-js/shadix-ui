## Usage

### Import the component

```tsx
import { DateTimePicker } from '@/components/datetimepicker';
```

### Basic Usage

```tsx
const DateTimePickerDemo = () => {
    return (
        <DateTimePicker
            mode="single"
            value={new Date()}
            onDateChange={(date) => {
                console.log(date);
            }}
            placeholder="Select date"
        />
    );
};
```

#### With Range Mode

```tsx
const DateTimePickerDemo = () => {
    return (
        <DateTimePicker
            mode="range"
            onDateChange={(date) => {
                console.log(date);
            }}
            placeholder="Select date range"
        />
    );
};
```

#### With Time Mode

```tsx
const DateTimePickerDemo = () => {
    return (
        <DateTimePicker
            mode="single"
            enableTime={true}
            timeFormat="24h" // 12h or 24h
            value={new Date()}
            onDateChange={(date) => {
                console.log(date);
            }}
            placeholder="Select date and time"
        />
    );
};
```
